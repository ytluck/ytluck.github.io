列表

#----创建--填充--添加--移除
##创建空列表，然后对列表进行数据添加
a <- rep(0:9)
K <- length(a)
lst <- list(NULL)
length(lst) <- 2*(K-1)


i <- 1
j <- 1
repeat{ 
        lst[[j]] <- c(a[0:i])
        lst[[j+1]] <- c(a[i+1:(K-i)])
        i <- i+1
        j <- j+2
        if(i> K-1) break
    }


 lst[[18]]   <- NULL
 lst[sapply(lst,is.null)] <- NULL
